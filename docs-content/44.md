## Full Hardening Audit

Audit one bucket against all 43 controls using every AWS CLI command in a single observation.

**AWS CLI commands**

Collect all configuration for one bucket. Save each output locally:

```bash
BUCKET=production-data

cat > policy.json <<'EOF'
{
  "Policy": "{\"Version\":\"2012-10-17\",\"Statement\":[{\"Sid\":\"AllowAppRead\",\"Effect\":\"Allow\",\"Principal\":{\"AWS\":\"arn:aws:iam::123456789012:role/AppReadRole\"},\"Action\":\"s3:GetObject\",\"Resource\":\"arn:aws:s3:::production-data/*\"}]}"
}
EOF

cat > public_access_block.json <<'EOF'
{
  "PublicAccessBlockConfiguration": {
    "BlockPublicAcls": true,
    "IgnorePublicAcls": true,
    "BlockPublicPolicy": true,
    "RestrictPublicBuckets": true
  }
}
EOF

cat > encryption.json <<'EOF'
{
  "ServerSideEncryptionConfiguration": {
    "Rules": [
      {
        "ApplyServerSideEncryptionByDefault": {
          "SSEAlgorithm": "aws:kms",
          "KMSMasterKeyID": "arn:aws:kms:us-east-1:123456789012:key/abcd1234-5678-90ab-cdef-1234567890ab"
        },
        "BucketKeyEnabled": true
      }
    ]
  }
}
EOF

cat > logging.json <<'EOF'
{
  "LoggingEnabled": {
    "TargetBucket": "access-logs-bucket",

  }

cat > acl.json <<'EOF'
```

**Extract to a Stave observation (`obs.v0.1`)**

One `jq` command builds a complete observation from all saved outputs:

```bash
mkdir -p observations

jq -n \
  --argjson policy "$(cat policy.json | jq '.Policy | fromjson')" \
  --argjson pab "$(cat public_access_block.json)" \
  --argjson enc "$(cat encryption.json)" \
  --argjson log "$(cat logging.json)" \
  --argjson ver "$(cat versioning.json)" \
  --argjson acl "$(cat acl.json)" \
  --argjson tags "$(cat tagging.json)" \
  --argjson lc "$(cat lifecycle.json)" \
  --argjson lock "$(cat lock.json)" \
  --argjson website "$(cat website.json)" \
  --argjson git "$(cat objects_git.json)" \
'{
  schema_version: "obs.v0.1",
  generated_by: { source_type: "aws-s3-snapshot", tool: "aws-cli", tool_version: "2.x" },
  captured_at: "2026-03-20T00:00:00Z",
  assets: [{
    id: "production-data",
    type: "aws_s3_bucket",
    vendor: "aws",
    properties: {
      storage: {
        kind: "bucket",
        name: "production-data",

        tags: ([$tags.TagSet[] | {(.Key): .Value}] | add // {}),

        controls: {
          public_access_fully_blocked: (
            $pab.PublicAccessBlockConfiguration |
            (.BlockPublicAcls == true and .IgnorePublicAcls == true
             and .BlockPublicPolicy == true and .RestrictPublicBuckets == true)
          )
        },

        access: {
          public_read: (
            [$policy.Statement[] |
              select(.Effect == "Allow" and (.Principal == "*" or .Principal.AWS == "*")
                and (.Action == "s3:GetObject" or .Action == "s3:*"
                  or ((.Action | type) == "array" and (.Action | any(. == "s3:GetObject" or . == "s3:*")))))
            ] | length > 0
          ),
          public_write: (
            [$policy.Statement[] |
              select(.Effect == "Allow" and (.Principal == "*" or .Principal.AWS == "*")
                and (.Action == "s3:PutObject" or .Action == "s3:DeleteObject" or .Action == "s3:*"
                  or ((.Action | type) == "array" and (.Action | any(. == "s3:PutObject" or . == "s3:DeleteObject" or . == "s3:*")))))
            ] | length > 0
          ),
          public_list: (
            [$policy.Statement[] |
              select(.Effect == "Allow" and (.Principal == "*" or .Principal.AWS == "*")
                and (.Action == "s3:ListBucket" or .Action == "s3:*"
                  or ((.Action | type) == "array" and (.Action | any(. == "s3:ListBucket" or . == "s3:*")))))
            ] | length > 0
          ),
          read_via_identity: (
            [$policy.Statement[] |
              select(.Effect == "Allow" and (.Principal == "*" or .Principal.AWS == "*")
                and (.Action == "s3:GetObject" or .Action == "s3:*"
                  or ((.Action | type) == "array" and (.Action | any(. == "s3:GetObject" or . == "s3:*")))))
            ] | length > 0
          ),
          list_via_identity: (
            [$policy.Statement[] |
              select(.Effect == "Allow" and (.Principal == "*" or .Principal.AWS == "*")
                and (.Action == "s3:ListBucket" or .Action == "s3:*"
                  or ((.Action | type) == "array" and (.Action | any(. == "s3:ListBucket" or . == "s3:*")))))
            ] | length > 0
          ),
          has_wildcard_policy: (
            [$policy.Statement[] |
              select(.Action == "s3:*" or .Action == "*"
                or ((.Action | type) == "array" and (.Action | any(. == "s3:*" or . == "*"))))
            ] | length > 0
          ),
          has_external_write: false,
          has_vpc_condition: false,
          has_ip_condition: false,
          effective_network_scope: "internet",
          external_account_ids: [],
          latent_public_read: false,
          latent_public_list: false,
          authenticated_read: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AuthenticatedUsers" and (.Permission == "READ" or .Permission == "FULL_CONTROL"))] | length > 0
          ),
          authenticated_write: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AuthenticatedUsers" and (.Permission == "WRITE" or .Permission == "FULL_CONTROL"))] | length > 0
          ),
          write_via_resource: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AllUsers" and (.Permission == "WRITE" or .Permission == "FULL_CONTROL"))] | length > 0
          )
        },

        acl: {
          public_full_control: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AllUsers" and .Permission == "FULL_CONTROL")] | length > 0
          ),
          public_read_acp: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AllUsers" and .Permission == "READ_ACP")] | length > 0
          ),
          public_write_acp: (
            [$acl.Grants[] | select(.Grantee.URI == "http://acs.amazonaws.com/groups/global/AllUsers" and .Permission == "WRITE_ACP")] | length > 0
          )
        },

        encryption: {
          at_rest_enabled: (
            ($enc.ServerSideEncryptionConfiguration.Rules // []) | length > 0
          ),
          in_transit_enforced: (
            [$policy.Statement[] |
              select(.Effect == "Deny" and .Condition.Bool["aws:SecureTransport"] == "false")
            ] | length > 0
          ),
          algorithm: (
            ($enc.ServerSideEncryptionConfiguration.Rules[0].ApplyServerSideEncryptionByDefault.SSEAlgorithm // null)
          ),
          kms_key_id: (
            ($enc.ServerSideEncryptionConfiguration.Rules[0].ApplyServerSideEncryptionByDefault.KMSMasterKeyID // null)
          )
        },

        logging: {
          enabled: ($log.LoggingEnabled != null)
        },

        versioning: {
          enabled: ($ver.Status == "Enabled"),
          mfa_delete: ($ver.MFADelete == "Enabled")
        },

        lifecycle: {
          rules_configured: (($lc.Rules // []) | length > 0),
          has_expiration: (
            [($lc.Rules // [])[] | select(.Status == "Enabled" and .Expiration.Days != null)] | length > 0
          ),
          min_expiration_days: (
            [($lc.Rules // [])[] | select(.Status == "Enabled" and .Expiration.Days != null) | .Expiration.Days] |
            if length == 0 then 0 else min end
          )
        },

        object_lock: {
          enabled: (($lock.ObjectLockConfiguration.ObjectLockEnabled // "") == "Enabled"),
          mode: ($lock.ObjectLockConfiguration.Rule.DefaultRetention.Mode // null),
          retention_days: (
            if $lock.ObjectLockConfiguration.Rule.DefaultRetention.Days != null
            then $lock.ObjectLockConfiguration.Rule.DefaultRetention.Days
            elif $lock.ObjectLockConfiguration.Rule.DefaultRetention.Years != null
            then $lock.ObjectLockConfiguration.Rule.DefaultRetention.Years * 365
            else 0 end
          )
        },

        website: {
          enabled: ($website.IndexDocument != null)
        },

        content: {
          exposed_repo_artifacts: (
            ($git.Contents // []) | any(.Key | test("^\\.(git|svn|env)"))
          )
        }
      }
    }
  }]
}' > observations/2026-03-20T000000Z.json

jq '.captured_at = "2026-03-21T00:00:00Z"' observations/2026-03-20T000000Z.json > observations/2026-03-21T000000Z.json
```

**Detect with Stave**

```bash
stave init --profile aws-s3

stave apply \
  --observations observations \
  --max-unsafe 12h \
  --now 2026-03-21T12:00:00Z
```

A well-hardened bucket should produce zero violations (exit code 0). Any findings indicate specific controls that need remediation — each finding references one of the 43 controls covered in articles 1–43.
